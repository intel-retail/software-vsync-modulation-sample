238 int do_secondary(
239  const char *server_name_or_ip_addr,
240  const char *ptp_eth_address,
241  int pipe,
242  int synchronize,
243  int timestamps,
244  double shift)
245 {
246  msg m, r;
247  int ret = 0;
248  long client_vsync[MAX_TIMESTAMPS], delta, avg;
249  long *va;
250  static int counter = 0;
251  pthread_t tid;
252  int status;
253  double avg_interval;
254 
255  if(ptp_eth_address) {
256  client = new ptp_connection(server_name_or_ip_addr, ptp_eth_address);
257  } else {
258  client = new connection();
259  }
260 
261  if(client->init_client(server_name_or_ip_addr)) {
262  delete client;
263  client = NULL;
264  return 1;
265  }
266 
267  if(timestamps > MAX_TIMESTAMPS) {
268  ERR("Number of vsync timestamps to collect can't be greater than or equal to %d", MAX_TIMESTAMPS);
269  delete client;
270  client = NULL;
271  return 1;
272  }
273 
274  do {
275  r.ack();
276  r.set_vblank_count(timestamps);
277  if(client->send_msg(&r, sizeof(r))) {
278  ret = 1;
279  }
280 
281  if(client->recv_msg(&m, sizeof(m))) {
282  ret = 1;
283  }
284  m.compare_time();
285 
286  } while(ret);
287  DBG("Received vsyncs from the primary system\n");
288 
289  if(get_vsync(client_vsync, timestamps)) {
290  delete client;
291  client = NULL;
292  return 1;
293  }
294 
295  va = m.get_va();
296 
297  print_vsyncs((char *) "PRIMARY'S", va, timestamps);
298  print_vsyncs((char *) "SECONDARY'S", client_vsync, timestamps);
299  delta = client_vsync[0] - va[timestamps-1];
300  avg = find_avg(va, timestamps);
301  DBG("Time average of the vsyncs on the primary system is %ld us\n", avg);
302  avg = find_avg(client_vsync, timestamps);
303  DBG("Time average of the vsyncs on the secondary system is %ld us\n", avg);
304  DBG("Time difference between secondary and primary is %ld us\n", delta);
305  /*
306  * If the primary is ahead or behind the secondary by more than a vsync,
307  * we can just adjust the secondary's vsync to what we think the primary's
308  * next vsync would be happening at. We do this by calculating the average
309  * of its last N vsyncs that it has provided to us and assuming that its
310  * next vsync would be happening at this cadence. Then we modulo the delta
311  * with this average to give us a time difference between it's nearest
312  * vsync to the secondary's. As long as we adjust the secondary's vsync
313  * to this value, it would basically mean that the primary and secondary
314  * system's vsyncs are firing at the same time.
315  */
316  if(delta > avg || delta < avg) {
317  delta %= avg;
318  }
319 
320  /*
321  * If the time difference between primary and secondary is larger than the
322  * mid point of secondary's vsync time period, then it makes sense to sync
323  * with the next vsync of the primary. For example, say primary's vsyncs
324  * are happening at a regular cadence of 0, 16.66, 33.33 ms while
325  * secondary's vsyncs are happening at a regular candence of 10, 26.66,
326  * 43.33 ms, then the time difference between them is exactly 10 ms. If
327  * we were to make the secondary faster for each iteration so that it
328  * walks back those 10 ms and gets in sync with the primary, it would have
329  * taken us about 600 iterations of vsyncs = 10 seconds. However, if were
330  * to make the secondary slower for each iteration so that it walks forward
331  * then since it is only 16.66 - 10 = 6.66 ms away from the next vsync of
332  * the primary, it would take only 400 iterations of vsyncs = 6.66 seconds
333  * to get in sync. Therefore, the general rule is that we should make
334  * secondary's vsyncs walk back only if it the delta is less than half of
335  * secondary's vsync time period, otherwise, we should walk forward.
336  */
337  if(delta > avg/2) {
338  delta -= avg;
339  }
340  INFO("Time difference between secondary and primary's next vsync is %ld us\n", delta);
341 
342  if(synchronize) {
343  if(abs(delta) > synchronize) {
344  avg_interval = get_vblank_interval(pipe);
345  INFO("VBlank interval before starting synchronization: %lf ms\n", avg_interval);
346 
347  thread_continue = 1;
348  // synchronize_vsync function is synchronous call and does not
349  // output any information. To enhance visibility, a thread is
350  // created for logging vblank intervals while synchronization is
351  // in progress.
352  status = pthread_create(&tid, NULL, background_task, &pipe);
353  if (status != 0) {
354  ERR("Cannot create thread");
355  return 1;
356  }
357 
358  synchronize_vsync((double) delta / 1000, pipe, shift);
359 
360  // Set flag to 0 to signal the thread to terminate
361  thread_continue = 0;
362  // Wait for the thread to terminate
363  pthread_join(tid, NULL);
364 
365  avg_interval = get_vblank_interval(pipe);
366  INFO("VBlank interval after synchronization ends: %lf ms\n", avg_interval);
367 
368  INFO("Synchronizing after %d seconds\n", counter);
369  counter = 0;
370  }
371  counter++;
372  }
373 
374  if(client) {
375  client->close_client();
376  delete client;
377  client = NULL;
378  }
379 
380  return ret;
381 }
