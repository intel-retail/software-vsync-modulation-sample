48 dkl::dkl(ddi_sel *ds, int first_dkl_phy_loc)
49 {
50  unsigned int val;
51 
52  if(ds->de_clk < first_dkl_phy_loc) {

56 
57  val = READ_OFFSET_DWORD(dkl_table[ds->de_clk - first_dkl_phy_loc].dkl_pll_div0.addr);

65  }

79 void dkl::program_phy(double time_diff, double shift)
80 {
81  ddi_sel *ds = get_ds();
82  dkl_phy_reg *dkl_phy = (dkl_phy_reg *) ds->phy_data;
83 
84 
85  dkl_phy->dkl_pll_div0.orig_val = dkl_phy->dkl_pll_div0.mod_val = READ_OFFSET_DWORD(dkl_phy->dkl_pll_div0.addr);
86 
87  /*
88  * Each Dekel PHY is addressed through a 4KB aperture. Each PHY has more than
89  * 4KB of register space, so a separate index is programmed in HIP_INDEX_REG0
90  * or HIP_INDEX_REG1, based on the port number, to set the upper 2 address
91  * bits that point the 4KB window into the full PHY register space.
92  * So, basically, if we read the value of the PLL DIV0 register and it is 0,
93  * then that means that this aperture may need to be shifted. Once we do that,
94  * we should re-read the value of the DIV0 register once more and this time it
95  * should be valid.
96  */
97  if(!dkl_phy->dkl_pll_div0.orig_val) {
98  WRITE_OFFSET_DWORD(dkl_phy->dkl_index.addr, dkl_phy->dkl_index_val);
99  dkl_phy->dkl_pll_div0.orig_val = dkl_phy->dkl_pll_div0.mod_val = READ_OFFSET_DWORD(dkl_phy->dkl_pll_div0.addr);
100  }
101 
102  dkl_phy->dkl_visa_serializer.orig_val = dkl_phy->dkl_visa_serializer.mod_val = READ_OFFSET_DWORD(dkl_phy->dkl_visa_serializer.addr);
103  dkl_phy->dkl_bias.orig_val = dkl_phy->dkl_bias.mod_val = READ_OFFSET_DWORD(dkl_phy->dkl_bias.addr);
104  dkl_phy->dkl_ssc.orig_val = dkl_phy->dkl_ssc.mod_val = READ_OFFSET_DWORD(dkl_phy->dkl_ssc.addr);
105  dkl_phy->dkl_dco.orig_val = dkl_phy->dkl_dco.mod_val = READ_OFFSET_DWORD(dkl_phy->dkl_dco.addr);
106  // For whichever PHY we find, let's set the done flag to 0 so that we can later
107  // have a timer for it to reset the default values back in their registers
108 
109 
110  if(time_diff < 0) {
111  shift *= -1;
112  }
113  int steps = CALC_STEPS_TO_SYNC(time_diff, shift);
114  DBG("steps are %d\n", steps);
115 
116  DBG("OLD VALUES\n dkl_pll_div0 [0x%X] =\t 0x%X\n dkl_visa_serializer [0x%X] =\t 0x%X\n "
117  "dkl_bias [0x%X] =\t 0x%X\n dkl_ssc [0x%X] =\t 0x%X\n dkl_dco [0x%X] =\t 0x%X\n",
118  dkl_phy->dkl_pll_div0.addr,
119  dkl_phy->dkl_pll_div0.orig_val,
120  dkl_phy->dkl_visa_serializer.addr,
121  dkl_phy->dkl_visa_serializer.orig_val,
122  dkl_phy->dkl_bias.addr,
123  dkl_phy->dkl_bias.orig_val,
124  dkl_phy->dkl_ssc.addr,
125  dkl_phy->dkl_ssc.orig_val,
126  dkl_phy->dkl_dco.addr,
127  dkl_phy->dkl_dco.orig_val);
128 
129  // PLL frequency in MHz (base) = 38.4* DKL_PLL_DIV0[i_fbprediv_3_0] *
130  // ( DKL_PLL_DIV0[i_fbdiv_intgr_7_0] + DKL_BIAS[i_fbdivfrac_21_0] / 2^22 )
131  int i_fbprediv_3_0 = GETBITS_VAL(dkl_phy->dkl_pll_div0.orig_val, 11, 8);
132  int i_fbdiv_intgr_7_0 = GETBITS_VAL(dkl_phy->dkl_pll_div0.orig_val, 7, 0);
133  int i_fbdivfrac_21_0 = GETBITS_VAL(dkl_phy->dkl_bias.orig_val, 29, 8);
134  double pll_freq = (double) (REF_CLK_FREQ * i_fbprediv_3_0 * (i_fbdiv_intgr_7_0 + i_fbdivfrac_21_0 / pow(2, 22)));
135  double new_pll_freq = pll_freq + (shift * pll_freq / 100);
136  double new_i_fbdivfrac_21_0 = ((new_pll_freq / (REF_CLK_FREQ * i_fbprediv_3_0)) - i_fbdiv_intgr_7_0) * pow(2,22);
137 
138  if(new_i_fbdivfrac_21_0 < 0) {
139  i_fbdiv_intgr_7_0 -= 1;
140  dkl_phy->dkl_pll_div0.mod_val &= ~GENMASK(7, 0);
141  dkl_phy->dkl_pll_div0.mod_val |= i_fbdiv_intgr_7_0;
142  new_i_fbdivfrac_21_0 = ((new_pll_freq / (REF_CLK_FREQ * i_fbprediv_3_0)) - (i_fbdiv_intgr_7_0)) * pow(2,22);
143  }
144 
145  DBG("old pll_freq \t %f\n", pll_freq);
146  DBG("new_pll_freq \t %f\n", new_pll_freq);
147  DBG("new fbdivfrac \t 0x%X\n", (int) new_i_fbdivfrac_21_0);
148 
149  dkl_phy->dkl_bias.mod_val &= ~GENMASK(29, 8);
150  dkl_phy->dkl_bias.mod_val |= (long)new_i_fbdivfrac_21_0 << 8;
151  dkl_phy->dkl_visa_serializer.mod_val &= ~GENMASK(2, 0);
152  dkl_phy->dkl_visa_serializer.mod_val |= 0x200;
153  dkl_phy->dkl_ssc.mod_val &= ~GENMASK(31, 29);
154  dkl_phy->dkl_ssc.mod_val |= 0x2 << 29;
155  dkl_phy->dkl_ssc.mod_val &= ~BIT(13);
156  dkl_phy->dkl_ssc.mod_val |= 0x2 << 12;
157  dkl_phy->dkl_dco.mod_val &= ~BIT(2);
158  dkl_phy->dkl_dco.mod_val |= 0x2 << 1;
159 
160  DBG("NEW VALUES\n dkl_pll_div0 [0x%X] =\t 0x%X\n dkl_visa_serializer [0x%X] =\t 0x%X\n "
161  "dkl_bias [0x%X] =\t 0x%X\n dkl_ssc [0x%X] =\t 0x%X\n dkl_dco [0x%X] =\t 0x%X\n",
162  dkl_phy->dkl_pll_div0.addr,
163  dkl_phy->dkl_pll_div0.mod_val,
164  dkl_phy->dkl_visa_serializer.addr,
165  dkl_phy->dkl_visa_serializer.mod_val,
166  dkl_phy->dkl_bias.addr,
167  dkl_phy->dkl_bias.mod_val,
168  dkl_phy->dkl_ssc.addr,
169  dkl_phy->dkl_ssc.mod_val,
170  dkl_phy->dkl_dco.addr,
171  dkl_phy->dkl_dco.mod_val);
172 
173  dkl_phy->done = 0;
174  user_info *ui = new user_info(this, dkl_phy);
175  make_timer((long) steps, ui, reset_phy_regs);
176 
177  program_mmio(dkl_phy, 1);
178 }

188 void dkl::wait_until_done()
189 {
190  TRACING();
191  ddi_sel *ds = get_ds();
192  dkl_phy_reg *dkl_phy = (dkl_phy_reg *) ds->phy_data;
193 
194  while(!dkl_phy->done && !lib_client_done) {
195  usleep(1000);
196  }
197 
198  // Restore original values in case of app termination

218 void dkl::program_mmio(dkl_phy_reg *pr, int mod)
219 {
220  /*
221  * Each Dekel PHY is addressed through a 4KB aperture. Each PHY has more than
222  * 4KB of register space, so a separate index is programmed in HIP_INDEX_REG0
223  * or HIP_INDEX_REG1, based on the port number, to set the upper 2 address
224  * bits that point the 4KB window into the full PHY register space.
225  * So, basically, if we read the value of the PLL DIV0 register and it is 0,
226  * then that means that this aperture may need to be shifted. Once we do that,
227  * we should re-read the value of the DIV0 register once more and this time it
228  * should be valid.
229  */
230  if(!READ_OFFSET_DWORD(pr->dkl_pll_div0.addr)) {
231  WRITE_OFFSET_DWORD(pr->dkl_index.addr, pr->dkl_index_val);
232  }
233 
234  WRITE_OFFSET_DWORD(pr->dkl_pll_div0.addr,
235  mod ? pr->dkl_pll_div0.mod_val : pr->dkl_pll_div0.orig_val);
236  WRITE_OFFSET_DWORD(pr->dkl_visa_serializer.addr,
237  mod ? pr->dkl_visa_serializer.mod_val : pr->dkl_visa_serializer.orig_val);
238  WRITE_OFFSET_DWORD(pr->dkl_bias.addr,
239  mod ? pr->dkl_bias.mod_val : pr->dkl_bias.orig_val);
240  WRITE_OFFSET_DWORD(pr->dkl_ssc.addr,
241  mod ? pr->dkl_ssc.mod_val : pr->dkl_ssc.orig_val);
242  WRITE_OFFSET_DWORD(pr->dkl_dco.addr,
243  mod ? pr->dkl_dco.mod_val : pr->dkl_dco.orig_val);
244 }

246 /*
247 * @brief
248 * This function resets the DKL Phy MMIO registers to their
249 * original value. It gets executed whenever a timer expires. We program MMIO
250 * registers of the PHY in this function becase we have waited for a certain
251 * time period to get the primary and secondary systems vsync in sync and now
252 * it is time to reprogram the default values for the secondary system's PHYs.
253 * @param sig - The signal that fired
254 * @param *si - A pointer to a siginfo_t, which is a structure containing
255 * further information about the signal
256 * @param *uc - This is a pointer to a ucontext_t structure, cast to void *.
257 * The structure pointed to by this field contains signal context information
258 * that was saved on the user-space stack by the kernel
259 * @return void
260 */
261 void dkl::reset_phy_regs(int sig, siginfo_t *si, void *uc)
262 {
263  user_info *ui = (user_info *) si->si_value.sival_ptr;
264  if(!ui) {
265  return;
266  }
267 
268  DBG("timer done\n");
269  dkl_phy_reg *dr = (dkl_phy_reg *) ui->get_reg();
270  dkl *d = (dkl *) ui->get_type();
271 
272  d->reset_phy_regs(dr);
273  delete ui;
274 }
