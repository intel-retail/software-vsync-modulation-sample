92 /*
93  * Like _PICK_EVEN(), but supports 2 ranges of evenly spaced address offsets.
94  * @__c_index corresponds to the index in which the second range starts to be
95  * used. Using math interval notation, the first range is used for indexes [ 0,
96  * @__c_index), while the second range is used for [ @__c_index, ... ). Example:
97  *
98  * #define _FOO_A 0xf000
99  * #define _FOO_B 0xf004
100  * #define _FOO_C 0xf008
101  * #define _SUPER_FOO_A 0xa000
102  * #define _SUPER_FOO_B 0xa100
103  * #define FOO(x) _MMIO(_PICK_EVEN_2RANGES(x, 3, \
104  * _FOO_A, _FOO_B, \
105  * _SUPER_FOO_A, _SUPER_FOO_B))
106  *
107  * This expands to:
108  * 0: 0xf000,
109  * 1: 0xf004,
110  * 2: 0xf008,
111  * 3: 0xa000,
112  * 4: 0xa100,
113  * 5: 0xa200,
114  * ...
115  */
116 #define _PICK_EVEN_2RANGES(__index, __c_index, __a, __b, __c, __d) \
117  (((__index) < (__c_index) ? _PICK_EVEN(__index, __a, __b) : \
118  _PICK_EVEN((__index) - (__c_index), __c, __d)))
