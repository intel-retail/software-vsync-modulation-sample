22 /*******************************************************************************
23  * Description
24  * calc_steps_to_sync - This function calculates how many steps we need to take
25  * in order to synchronize the primary and secondary systems given the delta
26  * between primary and secondary and the shift that we need to make in terms of
27  * percentage. Each steps is a single vsync period (typically 16.666 ms).
28  * Parameters
29  * double time_diff - The time difference in between the two systems in ms.
30  * double shift - The percentage shift that we need to make in our vsyncs.
31  * Return val
32  * int
33  ******************************************************************************/
34 int calc_steps_to_sync(double time_diff, double shift)
35 {
36  return (int) ((time_diff * 100) / shift);
