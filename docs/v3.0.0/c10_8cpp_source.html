39 c10::c10(ddi_sel *ds)
40 {
41 
42  /*
43  According to BSpecs, MTL and PTL DDIs for C10 correspond to ports A and B.
44  Consequently, it is reasonable to deduce the port number from de_clk as clk-1.
45  However, this assumption might not hold for future platforms, where it
46  may be necessary to derive the port number directly from register values.
47  For guidance on this approach, refer to combo.cpp.
48 
49  BSpecs:
50  Port Usage Capability Alternate Names
51  DDI A | Port A | eDP, DP, HDMI DDIA, Port A
52  DDI B | Port B | eDP, DP, HDMI DDIB, Port B
53  */
54  u32 port = ds->de_clk - 1;
55 
56  int pll_state;
57  for(int i = 0; i < C10_PLL_REG_COUNT; i++) {
58  pll_state = __intel_cx0_read(port, 0, PHY_C10_VDR_PLL(i));
59  DBG("c10: pll_state[%d] = 0x%X, %d\n",i, pll_state,pll_state);
60  }
61 
62  c10_reg.done=1;
63  ds->phy_data = &c10_reg;
64 
65  set_ds(ds);
66  set_init(true);
67 
68 }

94 void c10::program_phy(double time_diff, double shift)
95 {
96  ddi_sel *ds = get_ds();
97  c10_phy_reg *c10_phy = (c10_phy_reg *) ds->phy_data;
98 
99  int pll_state[C10_PLL_REG_COUNT];
100  int mpll_frac_rem_15_0, mpll_frac_den_15_0, mpll_frac_quot_15_0, mpll_multiplier_11_0, ref_clk_mpll_div_2_0;
101  int pll_freq;
102 
103  for(int i = 0; i < C10_PLL_REG_COUNT; i++) {
104  pll_state[i] = __intel_cx0_read(ds->de_clk - 1, 0, PHY_C10_VDR_PLL(i));
105  c10_phy->pll_state_old[i] = pll_state[i];
106  DBG("c10: pll_state[%d] = 0x%X, %d\n", i, pll_state[i],pll_state[i]);
107  }
108 
109 
110 
111  if(time_diff < 0) {
112  shift *= -1;
113  }
114 
115  int steps = CALC_STEPS_TO_SYNC(time_diff, shift);
116  DBG("steps are %d\n", steps);
117 
118  mpll_frac_den_15_0 = pll_state[C10_PLL_REG_DEN_HIGH] << 8 | pll_state[C10_PLL_REG_DEN_LOW];
119  mpll_frac_quot_15_0 = pll_state[C10_PLL_REG_QUOT_HIGH] << 8 | pll_state[C10_PLL_REG_QUOT_LOW];
120  mpll_frac_rem_15_0 = pll_state[C10_PLL_REG_REM_HIGH] << 8 | pll_state[C10_PLL_REG_REM_LOW];
121  mpll_multiplier_11_0 = (REG_FIELD_GET8(C10_PLL3_MULTIPLIERH_MASK, pll_state[C10_PLL_REG_MULTIPLIER]) << 8 |
122  pll_state[2]) / 2 + 16;
123  ref_clk_mpll_div_2_0 = REG_FIELD_GET8(C10_PLL15_TXCLKDIV_MASK, pll_state[C10_PLL_REG_TXCLKDIV]);
124 
125  pll_freq = DIV_ROUND_CLOSEST_ULL(mul_u32_u32(REF_CLK_FREQ * 1000, (mpll_multiplier_11_0 << 16) + mpll_frac_quot_15_0) +
126  DIV_ROUND_CLOSEST(REF_CLK_FREQ * 1000 * mpll_frac_rem_15_0, mpll_frac_den_15_0),
127  10 << (ref_clk_mpll_div_2_0 + 16));
128 
129  double new_pll_freq = pll_freq + (shift * pll_freq / 100);
130  unsigned long long temp = new_pll_freq * (10 << (ref_clk_mpll_div_2_0 + 16));
131  temp -= DIV_ROUND_CLOSEST(REF_CLK_FREQ * 1000 * mpll_frac_rem_15_0, mpll_frac_den_15_0);
132  temp /= (REF_CLK_FREQ * 1000);
133  temp -= ((unsigned long long)mpll_multiplier_11_0 << 16);
134  int new_mpll_frac_quot_15_0 = (int) temp;
135  c10_phy->pll_state[C10_PLL_REG_QUOT_LOW] = new_mpll_frac_quot_15_0 & GENMASK(7, 0);
136  c10_phy->pll_state[C10_PLL_REG_QUOT_HIGH] = new_mpll_frac_quot_15_0 >> 8;
137 
138  DBG("Old Quotient = 0x%X\n", mpll_frac_quot_15_0);
139  DBG("Denominator = 0x%X\n", mpll_frac_den_15_0);
140  DBG("Remainder = 0x%X\n", mpll_frac_rem_15_0);
141  DBG("Multiplier = 0x%X\n", mpll_multiplier_11_0);
142 
143  c10_phy->done = 0;
144  user_info *ui = new user_info(this, c10_phy);
145  make_timer((long) steps, ui, reset_phy_regs);
146 
147  program_mmio(c10_phy, 1);
148 }

231 {

234  u32 port = ds->de_clk - 1 ;
235 
236  intel_cx0_rmw(port, INTEL_CX0_BOTH_LANES, PHY_C10_VDR_CONTROL(1),

246  MB_WRITE_COMMITTED);

248

254  mod ? pr->pll_state[C10_PLL_REG_QUOT_HIGH] : pr->pll_state_old[C10_PLL_REG_QUOT_HIGH],

507 int c10::intel_cx0_wait_for_ack(u32 port, int command, int lane, u32 *val)
508 {
509  if (__intel_de_wait_for_register(
510  XELPDP_PORT_P2M_MSGBUS_STATUS(port, lane),
511  XELPDP_PORT_P2M_RESPONSE_READY,
512  XELPDP_PORT_P2M_RESPONSE_READY,
513  val)) {
514  ERR("Port %d Timeout waiting for message ACK. Status: 0x%x\n", port, *val);
515  return -ETIMEDOUT;
516  }
517 
518  if (*val & XELPDP_PORT_P2M_ERROR_SET) {
519  ERR("Port %d Error occurred during %s command. Status: 0x%x\n", port,
520  command == XELPDP_PORT_P2M_COMMAND_READ_ACK ? "read" : "write", *val);
521  intel_cx0_bus_reset(port, lane);
522  return -EINVAL;
523  }
524 
525  if (REG_FIELD_GET(XELPDP_PORT_P2M_COMMAND_TYPE_MASK, *val) != (u32) command) {
526  ERR("Port %d Not a %s response. MSGBUS Status: 0x%x.\n", port,
527  command == XELPDP_PORT_P2M_COMMAND_READ_ACK ? "read" : "write", *val);
528  intel_cx0_bus_reset(port, lane);
529  return -EINVAL;
530  }
531 
532  return 0;
533 }

543 int c10::__intel_cx0_read_once(u32 port, int lane, u16 addr)
544 {
545  int ack;
546  u32 val;
547 
548  if (intel_de_wait_for_clear(XELPDP_PORT_M2P_MSGBUS_CTL(port, lane),
549  XELPDP_PORT_M2P_TRANSACTION_PENDING,
550  XELPDP_MSGBUS_TIMEOUT_SLOW)) {
551  ERR("Port %d Timeout waiting for previous transaction to complete. Reset the bus and retry.\n", port);
552  intel_cx0_bus_reset(port, lane);
553  return -ETIMEDOUT;
554  }
555 
556  intel_de_write(XELPDP_PORT_M2P_MSGBUS_CTL(port, lane),
557  XELPDP_PORT_M2P_TRANSACTION_PENDING |
558  XELPDP_PORT_M2P_COMMAND_READ |
559  XELPDP_PORT_M2P_ADDRESS(addr));
560 
561  ack = intel_cx0_wait_for_ack(port, XELPDP_PORT_P2M_COMMAND_READ_ACK, lane, &val);
562  if (ack < 0) {
563  intel_cx0_bus_reset(port, lane);
564  return ack;
565  }
566 
567  intel_clear_response_ready_flag(port, lane);
568 
569  return REG_FIELD_GET(XELPDP_PORT_P2M_DATA_MASK, val);
570 }

609 int c10::__intel_cx0_write_once(u32 port, int lane, u16 addr, u8 data, bool committed)
610 {
611  u32 val;
612 
613  if (intel_de_wait_for_clear(XELPDP_PORT_M2P_MSGBUS_CTL(port, lane),
614  XELPDP_PORT_M2P_TRANSACTION_PENDING,
615  XELPDP_MSGBUS_TIMEOUT_SLOW)) {
616  ERR("Port %d Timeout waiting for previous transaction to complete. Resetting the bus.\n", port);
617  intel_cx0_bus_reset(port, lane);
618  return -ETIMEDOUT;
619  }
620 
621  intel_de_write(XELPDP_PORT_M2P_MSGBUS_CTL(port, lane),
622  XELPDP_PORT_M2P_TRANSACTION_PENDING |
623  (committed ? XELPDP_PORT_M2P_COMMAND_WRITE_COMMITTED :
624  XELPDP_PORT_M2P_COMMAND_WRITE_UNCOMMITTED) |
625  XELPDP_PORT_M2P_DATA(data) |
626  XELPDP_PORT_M2P_ADDRESS(addr));
627 
628  if (intel_de_wait_for_clear(XELPDP_PORT_M2P_MSGBUS_CTL(port, lane),
629  XELPDP_PORT_M2P_TRANSACTION_PENDING,
630  XELPDP_MSGBUS_TIMEOUT_SLOW)) {
631  ERR("Port %d Timeout waiting for write to complete. Resetting the bus.\n", port);
632  intel_cx0_bus_reset(port, lane);
633  return -ETIMEDOUT;
634  }
635 
636  if (committed) {
637  if (intel_cx0_wait_for_ack(port, XELPDP_PORT_P2M_COMMAND_WRITE_ACK, lane, &val) < 0) {
638  intel_cx0_bus_reset(port, lane);
639  return -EINVAL;
640  }
641  } else if ((intel_de_read(XELPDP_PORT_P2M_MSGBUS_STATUS(port, lane)) &
642  XELPDP_PORT_P2M_ERROR_SET)) {
643  ERR("Port %d Error occurred during write command.\n", port);
644  intel_cx0_bus_reset(port, lane);
645  return -EINVAL;
646  }
647 
648  intel_clear_response_ready_flag(port, lane);
649 
650  return 0;
651 }
