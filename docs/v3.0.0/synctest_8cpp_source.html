87 int main(int argc, char *argv[])
88 {
89  int ret = 0;

93 
94  INFO("Synctest Version: %s\n", get_version().c_str());
95 
96  int delta = 1000; // Drift in us to achieve from current time
97  double shift = 0.01;
98  int pipe = 0; // Default pipe# 0
99  int opt;
100  while ((opt = getopt(argc, argv, "p:d:s:v:h")) != -1) {
101  switch (opt) {
102  case 'p':
103  pipe = std::stoi(optarg);
104  break;
105  case 'd':
106  delta = std::stoi(optarg);
107  break;
108  case 's':
109  shift = std::stod(optarg);
110  break;
111  case 'v':
112  set_log_level(optarg);
113  break;
114  case 'h':
115  case '?':
116  if (optopt == 'p' || optopt == 'd' || optopt == 's' || optopt == 'v') {
117  ERR("Option -%c requires an argument.\n", char(optopt));
118  } else {
119  ERR("Unknown option: -%c\n", char(optopt));
120  }
121  print_help(argv[0]);
122  exit(EXIT_FAILURE);
123  }
124  }
125 
126  if(vsync_lib_init()) {
127  ERR("Couldn't initialize vsync lib\n");
128  return 1;
129  }
130 
131  avg_interval = get_vblank_interval(pipe);
132  INFO("VBlank interval before starting synchronization: %lf ms\n", avg_interval);
133 
134 
135  // synchronize_vsync function is synchronous call and does not
136  // output any information. To enhance visibility, a thread is
137  // created for logging vblank intervals while synchronization is
138  // in progress.
139  status = pthread_create(&tid, NULL, background_task, &pipe);
140  if (status != 0) {
141  ERR("Cannot create thread");
142  return 1;
143  }
144 
145  // Convert delta to milliseconds before calling
146  synchronize_vsync((double) delta / 1000.0, pipe, shift);
147  // Set flag to 0 to signal the thread to terminate
148  thread_continue = 0;
149  // Wait for the thread to terminate
150  pthread_join(tid, NULL);
151  avg_interval = get_vblank_interval(pipe);
152  INFO("VBlank interval after synchronization ends: %lf ms\n", avg_interval);
153 
154  vsync_lib_uninit();
155  return ret;
156 }
