73 {

79  INFO("Setting timer for %d ms\n", expire_ms);

130 int find_enabled_phys()
131 {
132  int i, j, val, ddi_select;
133  reg trans_ddi_func_ctl[] = {
134  REG(TRANS_DDI_FUNC_CTL_A),
135  REG(TRANS_DDI_FUNC_CTL_B),
136  REG(TRANS_DDI_FUNC_CTL_C),
137  REG(TRANS_DDI_FUNC_CTL_D),
138  };
139 
140  // According to the BSpec:
141  // 0000b None
142  // 0001b DDI A
143  // 0010b DDI B
144  // 0011b DDI C
145  // 0100b DDI USBC1
146  // 0101b DDI USBC2
147  // 0110b DDI USBC3
148  // 0111b DDI USBC4
149  // 1000b DDI USBC5
150  // 1001b DDI USBC6
151  // 1000b DDI D
152  // 1001b DDI E
153  // DISPLAY_CCU / DPCLKA Clock DE Internal Clock
154  // DDIA_DE_CLK DDIA
155  // DDIB_DE_CLK USBC1
156  // DDII_DE_CLK USBC2
157  // DDIJ_DE_CLK USBC3
158  // DDIK_DE_CLK USBC4
159 
160  phy_enabled_list = new list<phys *>;
161 
162  for(i = 0; i < ARRAY_SIZE(trans_ddi_func_ctl); i++) {
163  // First read the TRANS_DDI_FUNC_CTL to find if this pipe is enabled or not
164  val = READ_OFFSET_DWORD(trans_ddi_func_ctl[i].addr);
165  DBG("0x%X = 0x%X\n", trans_ddi_func_ctl[i].addr, val);
166  if(!(val & BIT(31))) {
167  DBG("Pipe %d is turned off\n", i+1);
168  continue;
169  }
170 
171 
172  // TRANS_DDI_FUNC_CTL bits 30:27 have the DDI which this pipe is connected to
173  ddi_select = GETBITS_VAL(val, 30, 27);
174  DBG("ddi_select = 0x%X\n", ddi_select);
175  for(j = 0; j < platform_table[supported_platform].ds_size; j++) {
176 
177  phys *new_phy = NULL;
178  // Match the DDI with the available ones on this platform
179  if(platform_table[supported_platform].ds[j].de_clk == ddi_select) {
180  switch(platform_table[supported_platform].ds[j].phy) {
181  case DKL:
182  DBG("Detected a DKL phy on pipe %d\n", i+1);
183  new_phy = new dkl(&platform_table[supported_platform].ds[j],
184  platform_table[supported_platform].first_dkl_phy_loc);
185  break;
186  case COMBO:
187  DBG("Detected a Combo phy on pipe %d\n", i+1);
188  new_phy = new combo(&platform_table[supported_platform].ds[j]);
189  break;
190  case C10:
191  DBG("Detected a C10 phy on pipe %d\n", i+1);
192  new_phy = new c10(&platform_table[supported_platform].ds[j]);
193  break;
194  case C20:
195  DBG("Detected a C20 phy on pipe %d. (Not implemented)\n", i+1);
196  break;
197  default:
198  ERR("Unsupported PHY. Phy is %d\n", platform_table[supported_platform].ds[j].phy);
199  return 1;
200  break;
201  }
202 
203  // Ignore PHYs that we don't support yet
204  if (new_phy == NULL) {
205  continue;
206  }
207  if(!new_phy->is_init()) {
208  ERR("PHY not initialized properly\n");
209  delete new_phy;
210  return 1;
211  }
212  new_phy->set_pipe(i);
213  phy_enabled_list->push_back(new_phy);
214 
215  // No point trying to find the same ddi if we have already found it once
216  break;
217  }
218  }
219  }
220  return 0;
221 }

259 void print_drm_info(void)
260 {
261  // This list covers most of the connector types that are supported by the DRM
262  const char* connector_type_str[] = {
263  "Unknown", // 0
264  "VGA", // 1
265  "DVI-I", // 2
266  "DVI-D", // 3
267  "DVI-A", // 4
268  "Composite", // 5
269  "S-Video", // 6
270  "LVDS", // 7
271  "Component", // 8
272  "9PinDIN", // 9
273  "DisplayPort", // 10
274  "HDMI-A", // 11
275  "HDMI-B", // 12
276  "TV", // 13
277  "eDP", // 14
278  "Virtual", // 15
279  "DSI", // 16
280  "DPI", // 17
281  "WriteBack", // 18
282  "SPI", // 19
283  "USB" // 20
284  };
285 
286  int fd = open_device();
287  if (fd < 0) {
288  ERR("Failed to open DRM device: %s\n", strerror(errno));
289  return ;
290  }
291 
292  drmModeRes *resources = drmModeGetResources(fd);
293  if (!resources) {
294  ERR("drmModeGetResources failed: %s\n", strerror(errno));
295  if(close(fd) == -1){
296  ERR("Failed to properly close file descriptor. Error: %s\n", strerror(errno));
297  }
298  return;
299  }
300  INFO("DRM Info:\n");
301 
302  // First print Pipe/CRTC info
303  INFO(" CRTCs found: %d\n", resources->count_crtcs);
304  for (int i = 0; i < resources->count_crtcs; i++) {
305  drmModeCrtc *crtc = drmModeGetCrtc(fd, resources->crtcs[i]);
306  if (!crtc) {
307  ERR("drmModeGetCrtc failed: %s\n", strerror(errno));
308  continue;
309  }
310 
311  double refresh_rate = 0;
312  if (crtc->mode_valid) {
313  refresh_rate = (double)crtc->mode.clock * 1000.0 / (crtc->mode.vtotal * crtc->mode.htotal);
314  }
315 
316  INFO(" \tPipe: %2d, CRTC ID: %4d, Mode Valid: %3s, Mode Name: %s, Position: (%4d, %4d), Resolution: %4dx%-4d, Refresh Rate: %.2f Hz\n",
317  i, crtc->crtc_id, (crtc->mode_valid) ? "Yes" : "No", crtc->mode.name,
318  crtc->x, crtc->y, crtc->mode.hdisplay, crtc->mode.vdisplay, refresh_rate);
319 
320  drmModeFreeCrtc(crtc);
321  }
322 
323  // Print Connector info
324  INFO(" Connectors found: %d\n", resources->count_connectors);
325  for (int i = 0; i < resources->count_connectors; i++) {
326  drmModeConnector *connector = drmModeGetConnector(fd, resources->connectors[i]);
327  if (!connector) {
328  ERR("drmModeGetConnector failed: %s\n", strerror(errno));
329  continue;
330  }
331 
332  INFO(" \tConnector: %-4d (ID: %-4d), Type: %-4d (%-12s), Type ID: %-4d, Connection: %-12s\n",
333  i, connector->connector_id, connector->connector_type, connector_type_str[connector->connector_type],
334  connector->connector_type_id,
335  (connector->connection == DRM_MODE_CONNECTED) ? "Connected" : "Disconnected");
336 
337  if (connector->connection == DRM_MODE_CONNECTED) {
338  drmModeEncoder *encoder = drmModeGetEncoder(fd, connector->encoder_id);
339  if (encoder) {
340  INFO("\t\t\tEncoder ID: %d, CRTC ID: %d\n", encoder->encoder_id, encoder->crtc_id);
341  drmModeFreeEncoder(encoder);
342  }
343  }
344  drmModeFreeConnector(connector);
345  }
346 
347  drmModeFreeResources(resources);
348  if(close(fd) == -1){
349  ERR("Failed to properly close file descriptor. Error: %s\n", strerror(errno));
350  }
351 }

363 int vsync_lib_init()
364 {
365  int device_id, i, j;
366  if(!IS_INIT()) {
367  // Show Pipe and CRTC info
368  print_drm_info();
369 
370  // Get the device id of this platform
371  device_id = get_device_id();
372  DBG("Device id is 0x%X\n", device_id);
373  // Loop through our supported platform list
374  for(i = 0; i < ARRAY_SIZE(platform_table); i++) {
375  for(j = 0; j < MAX_DEVICE_ID; j++) {
376  if(platform_table[i].device_ids[j] == device_id) {
377  break;
378  }
379  }
380  if(j != MAX_DEVICE_ID) {
381  break;
382  }
383  }
384  // This means we aren't on one of the supported platforms
385  if(i == ARRAY_SIZE(platform_table)) {
386  ERR("This platform is not supported. Device id is 0x%X\n", device_id);
387  return 1;
388  } else {
389  supported_platform = i;
390  }
391 
392  if(map_mmio()) {
393  return 1;
394  }
395 
396  if(find_enabled_phys()) {
397  vsync_lib_uninit();
398  return 1;
399  }
400 
401  INIT();
402  }
403 
404  return 0;
405 }

469 static void vblank_handler(int fd, unsigned int frame, unsigned int sec,
470  unsigned int usec, void *data)
471 {
472  drmVBlank vbl;
473  vbl_info *info = (vbl_info *)data;
474  memset(&vbl, 0, sizeof(drmVBlank));
475  if(info->counter < info->size) {
476  info->vsync_array[info->counter++] = TIME_IN_USEC(sec, usec);

482  vbl.request.signal = (unsigned long)data;
483 
484  drmWaitVBlank(g_dev_fd, &vbl);

525 int get_vsync(long *vsync_array, int size, int pipe)
526 {
527  drmVBlank vbl;
528  int ret;
529  drmEventContext evctx;
530  vbl_info handler_info;
531 
532  g_dev_fd = open_device();
533  if(g_dev_fd < 0) {
534  ERR("Couldn't open /dev/dri/card0. Is i915 installed?\n");
535  return -1;
536  }
537 
538  memset(&vbl, 0, sizeof(drmVBlank));
539 
540  handler_info.vsync_array = vsync_array;
541  handler_info.size = size;
542  handler_info.counter = 0;
543  handler_info.pipe = pipe;
544 
545  // Queue an event for frame + 1
546  vbl.request.type = (drmVBlankSeqType)
547  (DRM_VBLANK_RELATIVE | DRM_VBLANK_EVENT | pipe_to_wait_for(pipe));
548  DBG("vbl.request.type = 0x%X\n", vbl.request.type);
549  vbl.request.sequence = 1;
550  vbl.request.signal = (unsigned long)&handler_info;
551  ret = drmWaitVBlank(g_dev_fd, &vbl);
552  if (ret) {
553  ERR("drmWaitVBlank (relative, event) failed ret: %i\n", ret);
554  close_device();
555  return -1;
556  }
557 
558  // Set up our event handler
559  memset(&evctx, 0, sizeof evctx);
560  evctx.version = DRM_EVENT_CONTEXT_VERSION;
561  evctx.vblank_handler = vblank_handler;
562  evctx.page_flip_handler = NULL;
563 
564  // Poll for events
565  for(int i = 0; i < size; i++) {
566  struct timeval timeout = { .tv_sec = 3, .tv_usec = 0 };
567  fd_set fds;
568 
569  FD_ZERO(&fds);
570  FD_SET(0, &fds);
571  FD_SET(g_dev_fd, &fds);
572  ret = select(g_dev_fd + 1, &fds, NULL, NULL, &timeout);
573 
574  if (ret <= 0) {
575  ERR("select timed out or error (ret %d)\n", ret);
576  continue;
577  }
578 
579  ret = drmHandleEvent(g_dev_fd, &evctx);
580  if (ret) {
581  ERR("drmHandleEvent failed: %i\n", ret);
582  close_device();
583  return -1;
584  }
585  }
586 
587  close_device();
588  return 0;
589 }

598 double get_vblank_interval(int pipe)
599 {
600  const int MAX_STAMPS=20;
601  long timestamps[MAX_STAMPS];
602 
603  if (get_vsync(timestamps, MAX_STAMPS, pipe) == 0) {
604  long total_interval = 0;
605  for (int i = 0; i < MAX_STAMPS - 1; ++i) {
606  total_interval += (timestamps[i+1] - timestamps[i]);
607  }
608 
609  double avg_interval = total_interval / (MAX_STAMPS - 1) / 1000.0; // Convert to milliseconds
610  return avg_interval;
611  }
612  return 0.0;
