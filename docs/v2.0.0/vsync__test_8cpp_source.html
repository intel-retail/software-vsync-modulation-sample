115 long find_avg(long *va, int sz)
116 {
117  int avg = 0;

232 int do_secondary(
233  char *server_name_or_ip_addr,
234  char *ptp_eth_address,
235  int synchronize,
236  int timestamps)
237 {
238  msg m, r;
239  int ret = 0;
240  long client_vsync[MAX_TIMESTAMPS], delta, avg;
241  long *va;
242  static int counter = 0;
243 
244  if(ptp_eth_address) {
245  client = new ptp_connection(server_name_or_ip_addr, ptp_eth_address);
246  } else {
247  client = new connection();
248  }
249 
250  if(client->init_client(server_name_or_ip_addr)) {
251  delete client;
252  client = NULL;
253  return 1;
254  }
255 
256  if(timestamps > MAX_TIMESTAMPS) {
257  ERR("Number of vsync timestamps to collect can't be greater than or equal to %d", MAX_TIMESTAMPS);
258  delete client;
259  client = NULL;
260  return 1;
261  }
262 
263  do {
264  r.ack();
265  r.set_vblank_count(timestamps);
266  if(client->send_msg(&r, sizeof(r))) {
267  ret = 1;
268  }
269 
270  if(client->recv_msg(&m, sizeof(m))) {
271  ret = 1;
272  }
273  m.compare_time();
274 
275  } while(ret);
276  DBG("Received vsyncs from the primary system\n");
277 
278  if(get_vsync(client_vsync, timestamps)) {
279  delete client;
280  client = NULL;
281  return 1;
282  }
283 
284  va = m.get_va();
285 
286  print_vsyncs((char *) "PRIMARY'S", va, timestamps);
287  print_vsyncs((char *) "SECONDARY'S", client_vsync, timestamps);
288  delta = client_vsync[0] - va[timestamps-1];
289  avg = find_avg(va, timestamps);
290  DBG("Time average of the vsyncs on the primary system is %ld us\n", avg);
291  avg = find_avg(client_vsync, timestamps);
292  DBG("Time average of the vsyncs on the secondary system is %ld us\n", avg);
293  DBG("Time difference between secondary and primary is %ld us\n", delta);
294  /*
295  * If the primary is ahead or behind the secondary by more than a vsync,
296  * we can just adjust the secondary's vsync to what we think the primary's
297  * next vsync would be happening at. We do this by calculating the average
298  * of its last N vsyncs that it has provided to us and assuming that its
299  * next vsync would be happening at this cadence. Then we modulo the delta
300  * with this average to give us a time difference between it's nearest
301  * vsync to the secondary's. As long as we adjust the secondary's vsync
302  * to this value, it would basically mean that the primary and secondary
303  * system's vsyncs are firing at the same time.
304  */
305  if(delta > avg || delta < avg) {
306  delta %= avg;
307  }
308 
309  /*
310  * If the time difference between primary and secondary is larger than the
311  * mid point of secondary's vsync time period, then it makes sense to sync
312  * with the next vsync of the primary. For example, say primary's vsyncs
313  * are happening at a regular cadence of 0, 16.66, 33.33 ms while
314  * secondary's vsyncs are happening at a regular candence of 10, 26.66,
315  * 43.33 ms, then the time difference between them is exactly 10 ms. If
316  * we were to make the secondary faster for each iteration so that it
317  * walks back those 10 ms and gets in sync with the primary, it would have
318  * taken us about 600 iterations of vsyncs = 10 seconds. However, if were
319  * to make the secondary slower for each iteration so that it walks forward
320  * then since it is only 16.66 - 10 = 6.66 ms away from the next vsync of
321  * the primary, it would take only 400 iterations of vsyncs = 6.66 seconds
322  * to get in sync. Therefore, the general rule is that we should make
323  * secondary's vsyncs walk back only if it the delta is less than half of
324  * secondary's vsync time period, otherwise, we should walk forward.
325  */
326  if(delta > avg/2) {
327  delta -= avg;
328  }
329  INFO("Time difference between secondary and primary's next vsync is %ld us\n", delta);
330 
331  if(synchronize) {
332  if(abs(delta) > synchronize) {
333  synchronize_vsync((double) delta / 1000 );
334  INFO("Synchronizing after %d seconds\n", counter);
335  counter = 0;
336  }
337  counter++;
338  }
339 
340  if(client) {
341  client->close_client();
342  delete client;
343  client = NULL;
344  }
345 
346  return ret;
347 }

378  signal(SIGINT, client_close_signal);

393  us_synchronize = atoi(argv[4]);

395  do {

401  }
