41 combo::combo(ddi_sel *ds)
42 {
43  int dpclk;
44 
45  // Read DPCLKA_CFGCR0 or DPCLKA_CFGCR1 depending upon which DDI is enabled
46  dpclk = READ_OFFSET_DWORD(ds->dpclk.addr);
47  DBG("0x%X = 0x%X\n", ds->dpclk.addr, dpclk);
48  // DPCLKA_CFGCR must have the DDI's clock bit set to 0 (turned on)
49  if(dpclk & BIT(ds->clock_bit)) {
50  ERR("Clock is Off. dpclk = 0x%X, clock bit = 0x%X\n", dpclk, ds->clock_bit);
51  return;
52  }
53 
54  // Since there are only 2 bits for all DPLLs, it is safe to
55  // assume that the high order bit is just one more than the
56  // low order bit
57  ds->dpll_num = GETBITS_VAL(dpclk,
58  ds->mux_select_low_bit+1,
59  ds->mux_select_low_bit);
60  DBG("DPLL num = 0x%X\n", ds->dpll_num);
61 
62  if(ds->dpll_num >= ARRAY_SIZE(combo_table)) {
63  ERR("Dpll number (0x%X) is higher than combo table size.", ds->dpll_num);
64  return;
65  }
66  // One PHY should be connected to one display only
67  if(!combo_table[ds->dpll_num].enabled) {
68  combo_table[ds->dpll_num].enabled = 1;
69  ds->phy_data = &combo_table[ds->dpll_num];
70  set_ds(ds);
71  set_init(true);
72  }
73 }

83 int get_val_from_bit(div_val *dt, int dt_size, int bit)
84 {
85  int i, val = 0;

107 void combo::program_phy(double time_diff)
108 {
109  double shift = SHIFT;
110  div_val pdiv_table[4] = {
111  {1, 2},
112  {2, 3},
113  {4, 5},
114  {8, 7},
115  };
116 
117  div_val kdiv_table[3] = {
118  {1, 1},
119  {2, 2},
120  {4, 3},
121  };
122  ddi_sel *ds = get_ds();
123  combo_phy_reg *combo_phy = (combo_phy_reg *) ds->phy_data;
124 
125 #if TESTING
126  // ADL
127  /*
128  combo_phy->cfgcr0.orig_val = 0x01c001a5;
129  combo_phy->cfgcr1.orig_val = 0x013331cf;
130  */
131  // TGL
132  combo_phy->cfgcr0.orig_val = 0x00B001B1;
133  combo_phy->cfgcr1.orig_val = 0x00000e84;
134 #else
135  READ_VAL(cfgcr0, orig_val);
136  READ_VAL(cfgcr1, orig_val);
137 
138  if(time_diff < 0) {
139  shift *= -1;
140  }
141 
142  // For whichever PHY we find, let's set the done flag to 0 so that we can later
143  // have a timer for it to reset the default values back in their registers
144  combo_phy->done = 0;
145  int steps = CALC_STEPS_TO_SYNC(time_diff, shift);
146  DBG("steps are %d\n", steps);
147  user_info *ui = new user_info(this, combo_phy);
148  make_timer((long) steps, ui, reset_phy_regs);
149 #endif
150  DBG("OLD VALUES\n cfgcr0 [0x%X] =\t 0x%X\n cfgcr1 [0x%X] =\t 0x%X\n",
151  combo_phy->cfgcr0.addr, combo_phy->cfgcr0.orig_val,
152  combo_phy->cfgcr1.addr, combo_phy->cfgcr1.orig_val);
153 
154  // Symbol clock frequency in MHz (base) = DCO Divider// Reference frequency in MHz / (5// Pdiv// Qdiv// Kdiv)
155  // DCO Divider comes from DPLL_CFGCR0 DCO Integer + (DPLL_CFGCR0 DCO Fraction / 2^15)
156  // Pdiv from DPLL_CFGCR1 Pdiv
157  // Qdiv from DPLL_CFGCR1 Qdiv Mode ? DPLL_CFGCR1 Qdiv Ratio : 1
158  // Kdiv from DPLL_CFGCR1 Kdiv
159  int i_fbdiv_intgr_9_0 = GETBITS_VAL(combo_phy->cfgcr0.orig_val, 9, 0);
160  int i_fbdivfrac_14_0 = GETBITS_VAL(combo_phy->cfgcr0.orig_val, 24, 10);
161  int pdiv_bit = GETBITS_VAL(combo_phy->cfgcr1.orig_val, 5, 2);
162  int pdiv = get_val_from_bit(pdiv_table, ARRAY_SIZE(pdiv_table), pdiv_bit);
163  int kdiv_bit = GETBITS_VAL(combo_phy->cfgcr1.orig_val, 8, 6);
164  int kdiv = get_val_from_bit(kdiv_table, ARRAY_SIZE(kdiv_table), kdiv_bit);
165  int qdiv_mode = combo_phy->cfgcr1.orig_val & BIT(9);
166  /* TODO: In case we run into some other weird dividers, then we may need to revisit this */
167  int qdiv = (kdiv == 2 && qdiv_mode == 1) ? GETBITS_VAL(combo_phy->cfgcr1.orig_val, 17, 10) : 1;
168  int ro_div_bias_frac = i_fbdivfrac_14_0 << 5 | ((i_fbdiv_intgr_9_0 & GENMASK(2, 0)) << 19);
169  int ro_div_bias_int = i_fbdiv_intgr_9_0 >> 3;
170  double dco_divider = 4*((double) ro_div_bias_int + ((double) ro_div_bias_frac / pow(2,22)));
171  double dco_clock = 2 * REF_COMBO_FREQ * dco_divider;
172  double pll_freq = dco_clock / (5 * pdiv * qdiv * kdiv);
173  double new_pll_freq = pll_freq + (shift * pll_freq / 100);
174  double new_dco_clock = dco_clock + (shift * dco_clock / 100);
175  double new_dco_divider = new_dco_clock / (2 * REF_COMBO_FREQ);
176  double new_ro_div_bias_frac = (new_dco_divider / 4 - (double) ro_div_bias_int) * pow(2,22);
177  if(new_ro_div_bias_frac > 0xFFFFF) {
178  i_fbdiv_intgr_9_0 += 1;
179  new_ro_div_bias_frac -= 0xFFFFF;
180  } else if (new_ro_div_bias_frac < 0) {
181  i_fbdiv_intgr_9_0 -= 1;
182  new_ro_div_bias_frac += 0xFFFFF;
183  }
184  double new_i_fbdivfrac_14_0 = ((long int) new_ro_div_bias_frac & ~BIT(19)) >> 5;
185 
186  DBG("old pll_freq \t %f\n", pll_freq);
187  DBG("new_pll_freq \t %f\n", new_pll_freq);
188  DBG("old dco_clock \t %f\n", dco_clock);
189  DBG("new dco_clock \t %f\n", new_dco_clock);
190  DBG("old fbdivfrac \t 0x%X\n", (int) i_fbdivfrac_14_0);
191  DBG("old ro_div_frac \t 0x%X\n", ro_div_bias_frac);
192  DBG("old fbdivint \t 0x%X\n", i_fbdiv_intgr_9_0);
193  DBG("old ro_div_int \t 0x%X\n", ro_div_bias_int);
194  DBG("new fbdivfrac \t 0x%X\n", (int) new_i_fbdivfrac_14_0);
195  DBG("new ro_div_frac \t 0x%X\n", (int) new_ro_div_bias_frac);
196 
197  combo_phy->cfgcr0.mod_val &= ~GENMASK(24, 0);
198  combo_phy->cfgcr0.mod_val |= i_fbdiv_intgr_9_0;
199  combo_phy->cfgcr0.mod_val |= (long) new_i_fbdivfrac_14_0 << 10;
200 
201  DBG("NEW VALUES\n cfgcr0 [0x%X] =\t 0x%X\n", combo_phy->cfgcr0.addr,
202  combo_phy->cfgcr0.mod_val);
203  program_mmio(combo_phy, 1);
204 }
