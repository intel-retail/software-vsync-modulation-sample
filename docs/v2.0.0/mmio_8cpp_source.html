71 struct pci_device *intel_get_pci_device(void)
72 {
73  struct pci_device *pci_dev;
74  int error;
75 
76  error = pci_system_init();
77  if(error) {
78  ERR("Couldn't initialize PCI system\n");
79  return NULL;
80  }
81 
82  // Grab the graphics card. Try the canonical slot first, then
83  // walk the entire PCI bus for a matching device.
84  pci_dev = pci_device_find_by_slot(0, 0, 2, 0);
85  if (pci_dev == NULL || pci_dev->vendor_id != 0x8086) {
86  struct pci_device_iterator *iter;
87  struct pci_id_match match;
88 
89  match.vendor_id = 0x8086; // Intel
90  match.device_id = PCI_MATCH_ANY;
91  match.subvendor_id = PCI_MATCH_ANY;
92  match.subdevice_id = PCI_MATCH_ANY;
93 
94  match.device_class = 0x3 << 16;
95  match.device_class_mask = 0xff << 16;
96 
97  match.match_data = 0;
98 
99  iter = pci_id_match_iterator_create(&match);
100  pci_dev = pci_device_next(iter);
101  pci_iterator_destroy(iter);
102  }
103 
104  if(!pci_dev) {
105  ERR("Couldn't find Intel graphics card\n");
106  return NULL;
107  }
108 
109  error = pci_device_probe(pci_dev);
110  if(error) {
111  ERR("Couldn't probe graphics card\n");
112  return NULL;
113  }
114 
115  if (pci_dev->vendor_id != 0x8086) {
116  ERR("Graphics card is non-intel\n");
117  return NULL;
118  }
119 
120  return pci_dev;
121 }

192 int map_cmn(int base_index, int size)
193 {
194  std::lock_guard<std::mutex> lock(map_mutex); // Lock the mutex for the scope of the function
195 
196  int found = 0;
197  loff_t base;
198  int ret_val = 1;
199  unsigned long read_size = sizeof(gfx_pci_device);
200  const char proc_filename[] = "/proc/bus/pci/00/02.0";
201  struct stat file_stats;
202  FILE *fp;
203  unsigned char **temp;
204  size_t bytes_read;
205 
206  fp = fopen(proc_filename, "r");
207 
208  if(!fp) {
209 
210  ERR("Unable to open /proc/bus/pci/00/02.0\n");
211  ret_val = 0;
212 
213  } else {
214 
215  if(lstat(proc_filename, &file_stats) != 0) {
216 
217  ERR("Couldn't get file information for"
218  "/PROC/bus/pci/00/02.0\n");
219  ret_val = 0;
220 
221  } else {
222 
223  memset(g_raw_device, 0, 256);
224 
225  if((unsigned long) file_stats.st_size < sizeof(gfx_pci_device)) {
226  read_size = file_stats.st_size;
227  }
228 
229  bytes_read = fread(g_raw_device, 1, 256, fp);
230  if(bytes_read != 256) {
231  ERR("Couldn't read 256 bytes of data from PCI space\n");
232  ret_val = 0;
233  } else {
234  memcpy(&g_device, g_raw_device, read_size);
235  found = 1;
236  }
237  } /* end if(lstat...) */
238 
239  fclose(fp);
240 
241  if(!found) {
242  ret_val = 0;
243  } else {
244 
245  g_fd = open("/dev/mem",O_RDWR);
246  if(!g_fd) {
247  ERR("Could not open /dev/mem.\n");
248  ret_val = 0;
249  } else {
250 
251  base = (loff_t)(g_device.base_addr[base_index]&0xffffff80000);
252  if(base_index == 0) {
253  temp = &g_mmio;
254  g_mmio_size = size;
255  } else {
256  temp = &g_mem;
257  g_mem_size = size;
258  }
259  *temp = (unsigned char *) mmap(NULL,
260  size, PROT_READ | PROT_WRITE, MAP_SHARED,
261  g_fd, base);
262 
263  if(*temp == MAP_FAILED) {
264  ERR("Unable to mmap GMCH Registers: %s\n", strerror(errno));
265  ret_val = 0;
266  }
267  }
268  }
269  }
270  return ret_val;
271 }
