125 int find_enabled_phys()
126 {
127  int i, j, val, ddi_select;
128  reg trans_ddi_func_ctl[] = {
129  REG(TRANS_DDI_FUNC_CTL_A),
130  REG(TRANS_DDI_FUNC_CTL_B),
131  REG(TRANS_DDI_FUNC_CTL_C),
132  REG(TRANS_DDI_FUNC_CTL_D),
133  };
134  phys *new_phy;
135 
136  // According to the BSpec:
137  // 0000b None
138  // 0001b DDI A
139  // 0010b DDI B
140  // 0011b DDI C
141  // 0100b DDI USBC1
142  // 0101b DDI USBC2
143  // 0110b DDI USBC3
144  // 0111b DDI USBC4
145  // 1000b DDI USBC5
146  // 1001b DDI USBC6
147  // 1000b DDI D
148  // 1001b DDI E
149  // DISPLAY_CCU / DPCLKA Clock DE Internal Clock
150  // DDIA_DE_CLK DDIA
151  // DDIB_DE_CLK USBC1
152  // DDII_DE_CLK USBC2
153  // DDIJ_DE_CLK USBC3
154  // DDIK_DE_CLK USBC4
155 
156  phy_enabled_list = new list<phys *>;
157 
158  for(i = 0; i < ARRAY_SIZE(trans_ddi_func_ctl); i++) {
159  // First read the TRANS_DDI_FUNC_CTL to find if this pipe is enabled or not
160  val = READ_OFFSET_DWORD(trans_ddi_func_ctl[i].addr);
161  DBG("0x%X = 0x%X\n", trans_ddi_func_ctl[i].addr, val);
162  if(!(val & BIT(31))) {
163  DBG("Pipe %d is turned off\n", i+1);
164  continue;
165  }
166 
167  // TRANS_DDI_FUNC_CTL bits 30:27 have the DDI which this pipe is connected to
168  ddi_select = GETBITS_VAL(val, 30, 27);
169  DBG("ddi_select = 0x%X\n", ddi_select);
170  for(j = 0; j < platform_table[supported_platform].ds_size; j++) {
171  // Match the DDI with the available ones on this platform
172  if(platform_table[supported_platform].ds[j].de_clk == ddi_select) {
173  switch(platform_table[supported_platform].ds[j].phy) {
174  case DKL:
175  DBG("Detected a DKL phy on pipe %d\n", i+1);
176  new_phy = new dkl(&platform_table[supported_platform].ds[j],
177  platform_table[supported_platform].first_dkl_phy_loc);
178  break;
179  case COMBO:
180  DBG("Detected a Combo phy on pipe %d\n", i+1);
181  new_phy = new combo(&platform_table[supported_platform].ds[j]);
182  break;
183  default:
184  ERR("Unsupported PHY. Phy is %d\n", platform_table[supported_platform].ds[j].phy);
185  return 1;
186  break;
187  }
188  if(!new_phy->is_init()) {
189  ERR("PHY not initialized properly\n");
190  delete new_phy;
191  return 1;
192  }
193  new_phy->set_pipe(i);
194  phy_enabled_list->push_back(new_phy);
195 
196  // No point trying to find the same ddi if we have already found it once
197  break;
198  }
199  }
200  }
201  return 0;
202 }

232 int vsync_lib_init()
233 {
234  int device_id, i, j;
235  if(!IS_INIT()) {
236  // Get the device id of this platform
237  device_id = get_device_id();
238  DBG("Device id is 0x%X\n", device_id);
239  // Loop through our supported platform list
240  for(i = 0; i < ARRAY_SIZE(platform_table); i++) {
241  for(j = 0; j < MAX_DEVICE_ID; j++) {
242  if(platform_table[i].device_ids[j] == device_id) {
243  break;
244  }
245  }
246  if(j != MAX_DEVICE_ID) {
247  break;
248  }
249  }
250  // This means we aren't on one of the supported platforms
251  if(i == ARRAY_SIZE(platform_table)) {
252  ERR("This platform is not supported. Device id is 0x%X\n", device_id);
253  return 1;
254  } else {
255  supported_platform = i;
256  }
257 
258  if(map_mmio()) {
259  return 1;
260  }
261 
262  if(find_enabled_phys()) {
263  vsync_lib_uninit();
264  return 1;
265  }
266 
267  INIT();
268  }
269 
270  return 0;
271 }

334 static void vblank_handler(int fd, unsigned int frame, unsigned int sec,
335  unsigned int usec, void *data)
336 {
337  drmVBlank vbl;
338  vbl_info *info = (vbl_info *)data;
339  memset(&vbl, 0, sizeof(drmVBlank));
340  if(info->counter < info->size) {
341  info->vsync_array[info->counter++] = TIME_IN_USEC(sec, usec);
342  }
343 
344  vbl.request.type = (drmVBlankSeqType) (DRM_VBLANK_RELATIVE | DRM_VBLANK_EVENT |
345  pipe_to_wait_for(info->pipe));
346  vbl.request.sequence = 1;
347  vbl.request.signal = (unsigned long)data;
348 
349  drmWaitVBlank(g_dev_fd, &vbl);
350 }

366 unsigned int pipe_to_wait_for(int pipe)
367 {
368  int ret = 0;
369  if (pipe > 1) {
370  ret = (pipe << DRM_VBLANK_HIGH_CRTC_SHIFT) & DRM_VBLANK_HIGH_CRTC_MASK;
371  } else if (pipe > 0) {
372  ret = DRM_VBLANK_SECONDARY;
373  }
374  return ret;
375 }

390 int get_vsync(long *vsync_array, int size, int pipe)
391 {
392  drmVBlank vbl;
393  int ret;
394  drmEventContext evctx;
395  vbl_info handler_info;
396 
397  g_dev_fd = open_device();
398  if(g_dev_fd < 0) {
399  ERR("Couldn't open /dev/dri/card0. Is i915 installed?\n");
400  return -1;
401  }
402 
403  memset(&vbl, 0, sizeof(drmVBlank));
404 
405  handler_info.vsync_array = vsync_array;
406  handler_info.size = size;
407  handler_info.counter = 0;
408  handler_info.pipe = pipe;
409 
410  // Queue an event for frame + 1
411  vbl.request.type = (drmVBlankSeqType)
412  (DRM_VBLANK_RELATIVE | DRM_VBLANK_EVENT | pipe_to_wait_for(pipe));
413  DBG("vbl.request.type = 0x%X\n", vbl.request.type);
414  vbl.request.sequence = 1;
415  vbl.request.signal = (unsigned long)&handler_info;
416  ret = drmWaitVBlank(g_dev_fd, &vbl);
417  if (ret) {
418  ERR("drmWaitVBlank (relative, event) failed ret: %i\n", ret);
419  close_device();
420  return -1;
421  }
422 
423  // Set up our event handler
424  memset(&evctx, 0, sizeof evctx);
425  evctx.version = DRM_EVENT_CONTEXT_VERSION;
426  evctx.vblank_handler = vblank_handler;
427  evctx.page_flip_handler = NULL;
428 
429  // Poll for events
430  for(int i = 0; i < size; i++) {
431  struct timeval timeout = { .tv_sec = 3, .tv_usec = 0 };
432  fd_set fds;
433 
434  FD_ZERO(&fds);
435  FD_SET(0, &fds);
436  FD_SET(g_dev_fd, &fds);
437  ret = select(g_dev_fd + 1, &fds, NULL, NULL, &timeout);
438 
439  if (ret <= 0) {
440  ERR("select timed out or error (ret %d)\n", ret);
441  continue;
442  }
443 
444  ret = drmHandleEvent(g_dev_fd, &evctx);
445  if (ret) {
446  ERR("drmHandleEvent failed: %i\n", ret);
447  close_device();
448  return -1;
449  }
450  }
451 
452  close_device();
453  return 0;
454 }
